select
PD.PERSONNEL_NUMBER,
PD.FIRST_NAME,
PD.LAST_NAME,
PD.FORMATTED_NAME,
(select description from SAP_OBJECT_DESCRIPTIONS where OBJECT_TYPE = POSITION_OBJECT_TYPE AND OBJECT_ID= POSITION_OBJECT_ID) POSITION,
PD.PERSONNEL_SUBAREA_AND_DES,
PD.NETWORK_USER_ID,
PD.EMAIL_ADDRESS,
PD.MOBILE_PHONE_NUMBER,
PD.BUSINESS_PHONE_NUMBER,
PD.FAX_PHONE_NUMBER,
PD.SUPERVISOR_USER_NAME,
PD.COMPANY_NAME,
DECODE((SELECT 'X' FROM DUAL WHERE EXISTS( SELECT 'X' FROM SAP_PERSONNEL_QUAL_DETAILS T_QUAL_MAF WHERE T_QUAL_MAF.qual_object_id = 50003330 and T_QUAL_MAF.personnel_number = PD.PERSONNEL_NUMBER)), 'X','Y','N')  QUAL_MAF,
DECODE((SELECT 'X' FROM DUAL WHERE EXISTS( SELECT 'X' FROM SAP_PERSONNEL_QUAL_DETAILS T_QUAL_DS WHERE T_QUAL_DS.qual_object_id = 50003451 and T_QUAL_DS.personnel_number = PD.PERSONNEL_NUMBER)), 'X','Y','N')  QUAL_DS,
DECODE((SELECT 'X' FROM DUAL WHERE EXISTS( SELECT 'X' FROM SAP_PERSONNEL_QUAL_DETAILS T_QUAL_RSO WHERE T_QUAL_RSO.qual_object_id = 50003446 and T_QUAL_RSO.personnel_number = PD.PERSONNEL_NUMBER)), 'X','Y','N')  QUAL_RSO,
DECODE((SELECT 'X' FROM DUAL WHERE EXISTS( SELECT 'X' FROM SAP_PERSONNEL_QUAL_DETAILS T_QUAL_TSF WHERE T_QUAL_TSF.qual_object_id = 50003449 and T_QUAL_TSF.personnel_number = PD.PERSONNEL_NUMBER)), 'X','Y','N')  QUAL_TSF,
DECODE((SELECT 'X' FROM DUAL WHERE EXISTS( SELECT 'X' FROM SAP_PERSONNEL_QUAL_DETAILS T_QUAL_ZSS WHERE T_QUAL_ZSS.qual_object_id = 50003448 and T_QUAL_ZSS.personnel_number = PD.PERSONNEL_NUMBER)), 'X','Y','N')  QUAL_ZSS,
DECODE((SELECT 'X' FROM DUAL WHERE EXISTS( SELECT 'X' FROM SAP_PERSONNEL_QUAL_DETAILS T_QUAL_DSO WHERE T_QUAL_DSO.qual_object_id = 50003445 and T_QUAL_DSO.personnel_number = PD.PERSONNEL_NUMBER)), 'X','Y','N')  QUAL_DSO
from SAP_PERSONNEL_DETAILS PD
